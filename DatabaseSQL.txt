-- SMART RESEARCH PUBLICATION MANAGEMENT DATABASE
-- Portable Version - No Hardcoded Schema
------------------------------------------------------------

-----------------------------
-- SEQUENCES
-----------------------------
CREATE SEQUENCE USER_SEQ START WITH 1 INCREMENT BY 1;
CREATE SEQUENCE SUBMISSION_SEQ START WITH 1 INCREMENT BY 1;
CREATE SEQUENCE PUB_SEQ START WITH 1 INCREMENT BY 1;
CREATE SEQUENCE CONF_SEQ START WITH 1 INCREMENT BY 1;
CREATE SEQUENCE REVIEW_SEQ START WITH 1 INCREMENT BY 1;
CREATE SEQUENCE COLLAB_SEQ START WITH 1 INCREMENT BY 1;

-----------------------------
-- TABLES
-----------------------------

CREATE TABLE USERS (
    USERID NUMBER PRIMARY KEY,
    EMAIL VARCHAR2(100),
    PASSWORD VARCHAR2(256),
    USERNAME VARCHAR2(100),
    ROLE VARCHAR2(50),
    SPECIALTY VARCHAR2(50)
);

CREATE TABLE SUBMISSION (
    SUBMISSIONID NUMBER PRIMARY KEY,
    TITLE VARCHAR2(255) NOT NULL,
    STATUS VARCHAR2(20) NOT NULL,
    SOURCESCITATION CLOB,
    CREATEDAT DATE DEFAULT SYSDATE,
    AUTHORID NUMBER NOT NULL,
    TOPIC VARCHAR2(255) NOT NULL,
    MANUSCRIPT CLOB,
    CONSTRAINT CHK_SUBMISSION_STATUS CHECK (STATUS IN (
        'Draft',
        'Submitted',
        'Under Review',
        'Revision Required',
        'Accepted',
        'Rejected'
    ))
);

CREATE TABLE PUBLICATIONS (
    PUBLICATIONID NUMBER PRIMARY KEY,
    AUTHORID NUMBER NOT NULL,
    SUBMISSIONID NUMBER NOT NULL,
    DESCRIPTION VARCHAR2(300),
    CREATEDDATE DATE,
    FIELD VARCHAR2(50)
);

CREATE TABLE CONFERENCE (
    ID NUMBER PRIMARY KEY,
    TITLE VARCHAR2(200),
    PUBLICATIONID NUMBER,
    CONF_DATE DATE,
    LOCATION VARCHAR2(200),
    PRICE NUMBER
);

CREATE TABLE REVIEW (
    IDREVIEW NUMBER PRIMARY KEY,
    REVIEWER_NAME VARCHAR2(255) NOT NULL,
    REVIEW_DATE DATE NOT NULL,
    SUBMISSION_ID NUMBER NOT NULL,
    PUBLICATION_ID NUMBER NOT NULL,
    STATUS VARCHAR2(50) DEFAULT 'Pending',
    COMMENTREVIEW VARCHAR2(1000)
);

CREATE TABLE COLLABORATION (
    IDCOLLABORATION NUMBER PRIMARY KEY,
    TITLE VARCHAR2(50),
    DESCRIPTION VARCHAR2(600),
    AUTHORID NUMBER,
    PUBLICATIONID NUMBER
);

CREATE TABLE USER_SUBMISSION (
    USERID NUMBER,
    SUBMISSIONID NUMBER,
    PRIMARY KEY (USERID, SUBMISSIONID)
);

-----------------------------
-- FOREIGN KEYS
-----------------------------

ALTER TABLE SUBMISSION
ADD CONSTRAINT FK_SUBMISSION_AUTHOR
FOREIGN KEY (AUTHORID) REFERENCES USERS(USERID);

ALTER TABLE PUBLICATIONS
ADD CONSTRAINT FK_PUBLICATION_AUTHOR
FOREIGN KEY (AUTHORID) REFERENCES USERS(USERID);

ALTER TABLE PUBLICATIONS
ADD CONSTRAINT FK_PUBLICATION_SUBMISSION
FOREIGN KEY (SUBMISSIONID) REFERENCES SUBMISSION(SUBMISSIONID);

ALTER TABLE CONFERENCE
ADD CONSTRAINT FK_CONFERENCE_PUBLICATION
FOREIGN KEY (PUBLICATIONID) REFERENCES PUBLICATIONS(PUBLICATIONID);

ALTER TABLE REVIEW
ADD CONSTRAINT FK_REVIEW_SUBMISSION
FOREIGN KEY (SUBMISSION_ID) REFERENCES SUBMISSION(SUBMISSIONID);

ALTER TABLE REVIEW
ADD CONSTRAINT FK_REVIEW_PUBLICATION
FOREIGN KEY (PUBLICATION_ID) REFERENCES PUBLICATIONS(PUBLICATIONID);

ALTER TABLE COLLABORATION
ADD CONSTRAINT FK_COLLAB_AUTHOR
FOREIGN KEY (AUTHORID) REFERENCES USERS(USERID);

ALTER TABLE COLLABORATION
ADD CONSTRAINT FK_COLLAB_PUBLICATION
FOREIGN KEY (PUBLICATIONID) REFERENCES PUBLICATIONS(PUBLICATIONID);

ALTER TABLE USER_SUBMISSION
ADD CONSTRAINT FK_USER_SUB
FOREIGN KEY (USERID) REFERENCES USERS(USERID);

ALTER TABLE USER_SUBMISSION
ADD CONSTRAINT FK_SUB_USER
FOREIGN KEY (SUBMISSIONID) REFERENCES SUBMISSION(SUBMISSIONID);

ALTER TABLE USERS ADD CONSTRAINT chk_email CHECK (EMAIL LIKE '%@%.%');

ALTER TABLE USERS ADD CONSTRAINT pass_len CHECK (LENGTH(PASSWORD) >= 6);

ALTER TABLE USERS ADD CONSTRAINT chk_username CHECK ( REGEXP_LIKE(USERNAME, '^[A-Za-z]+( [A-Za-z]+){0,2}$'));

ALTER TABLE USERS ADD CONSTRAINT chk_specialty CHECK ( REGEXP_LIKE( SPECIALTY, '^[A-Za-z]+( [A-Za-z]+){0,2}$'));

-- Add RFID to users
ALTER TABLE USERS ADD RFID_CODE VARCHAR2(50) UNIQUE;

-- Labs sequence
CREATE SEQUENCE LAB_SEQ START WITH 1 INCREMENT BY 1;

-- Labs table
CREATE TABLE LABS (
    LABID       NUMBER PRIMARY KEY,
    LAB_NAME    VARCHAR2(100) NOT NULL,
    NORMAL_TEMP NUMBER(5,2)   NOT NULL,
    IS_OCCUPIED NUMBER(1)     DEFAULT 0 NOT NULL,
    CONSTRAINT CHK_LAB_OCCUPIED CHECK (IS_OCCUPIED IN (0, 1))
);
